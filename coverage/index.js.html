    return a+b;

/*
 Задание 3:
 
 Функция должна возвращать все переданные в нее аргументы в виде массива
 Количество переданных аргументов заранее неизвестно
 */
function returnArgumentsArray() {
    var arg = [];
    for( var i = 0; i < arguments.length; i++ ) {
        arg[i] = arguments[i];
    }
    return arg;
}

        return ++number;
    }
}
